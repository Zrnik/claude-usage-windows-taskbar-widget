4. Přidat `OnRightClick` event handler:

private void OnRightClick(object sender, MouseButtonEventArgs e)
{
    var menu = new ContextMenu();
    var quit = new MenuItem { Header = "Quit" };
    quit.Click += (_, _) => Application.Current.Shutdown();
    menu.Items.Add(quit);
    menu.Placement = System.Windows.Controls.Primitives.PlacementMode.MousePoint;
    menu.IsOpen = true;
}

Upravit `App.OnStartup` v Program.cs:

```csharp
protected override void OnStartup(StartupEventArgs e)
{
    base.OnStartup(e);
    ShutdownMode = ShutdownMode.OnMainWindowClose;

    var credential = CredentialStore.LoadCredential();
    if (credential == null)
    {
        MessageBox.Show("Claude credentials not found.\nExpected: ~/.claude/.credentials.json",
            "Claude Usage Widget", MessageBoxButton.OK, MessageBoxImage.Error);
        Shutdown();
        return;
    }

    var apiClient = new ClaudeApiClient();
    apiClient.SetCredential(credential);

    Exit += (_, _) => apiClient.Dispose();

    var window = new MainWindow(apiClient);
    MainWindow = window;
    window.Show();
}
```
